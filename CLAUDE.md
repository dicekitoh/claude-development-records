from googleapiclient.discovery import build
from google.auth.transport.requests import Request

# トークン読み込み（Calendar APIスコープ付き）
token_file = '/home/rootmax/google_extended_all_apis_token.pickle'
with open(token_file, 'rb') as f:
    creds = pickle.load(f)

# Calendar API接続

# 新しい項目（ここを変更）
NEW_ITEM = 'ここに新しい検討項目を入力'

# カレンダー検索・項目追加処理
calendars = service.calendarList().list().execute()
routine_cal_id = None
for cal in calendars.get('items', []):
    if 'ルーティンワーク' in cal.get('summary', ''):
        routine_cal_id = cal['id']
        break

if routine_cal_id:
    events = service.events().list(
        calendarId=routine_cal_id,
        q='予実管理',
        maxResults=250
    ).execute()
    
    updated = 0
    for event in events.get('items', []):
        if '予実管理' in event.get('summary', ''):
            desc = event.get('description', '')
            if desc:
                lines = desc.strip().split('\\n')
                last_num = len(lines)
                new_desc = desc + f'\\n{last_num + 1}. {NEW_ITEM}'
            else:
                new_desc = f'1. {NEW_ITEM}'
            
            event['description'] = new_desc
            service.events().update(
                calendarId=routine_cal_id,
                eventId=event['id'],
                body=event
            ).execute()
            updated += 1
    
    print(f'✅ {updated}件の予実管理イベントに「{NEW_ITEM}」を追加しました')
else:
    print('❌ ルーティンワークカレンダーが見つかりません')
